for here-docs : int main(void)
{
	char *prompt;
	int		fd2 = open("file2.txt", O_RDWR | O_CREAT | O_APPEND, 0644);
	char *delimiter = "eof";
	while(1)
	{
		prompt = readline(">");
		if (ft_strcmp(prompt, delimiter) == 0)
			return (1);
		else
		{
			ft_putstr_fd(prompt, fd2);
			write(fd2, "\n", 1);
		}
		free(prompt);
	}
	close(fd2);
	return (0);
}
